FROM openjdk:20-jdk-bullseye

WORKDIR /app
RUN apt-get update && apt-get install dos2unix
COPY . .

RUN find . -type f -print0 | xargs -0 dos2unix

RUN ["./mvnw", "package", "-DskipTests"]

CMD ["java", "-jar", "-Dspring.profiles.active=prod", "target/sporty-shoes-0.0.1-SNAPSHOT.jar"]


version: '3.8'
services:
  db:
    image: mysql:8.0
    environment:
      - MYSQL_DATABASE=${DB_NAME}
      - MYSQL_ROOT_PASSWORD=${DB_PASSWORD}
    volumes:
      - db:/var/lib/mysql
      - ./db/init.sql:/docker-entrypoint-initdb.d/init.sql
    ports:
      - '3306:3306'
  backend:
    depends_on:
      - db
    build:
      context: ./
      dockerfile: Dockerfile
    ports:
      - '80:8080'
    environment:
      - DB_NAME=${DB_NAME}
      - DB_USER=${DB_USER}
      - DB_PASSWORD=${DB_PASSWORD}
volumes:
  db:
    driver: local

#!/bin/bash
echo Installing Docker

sudo apt-get --yes remove docker docker.io containerd runc
sudo apt-get --yes update
sudo apt-get --yes install \
    ca-certificates \
    curl \
    gnupg \
    lsb-release


sudo mkdir -p /etc/apt/keyrings
curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /etc/apt/keyrings/docker.gpg
sudo chmod a+r /etc/apt/keyrings/docker.gpg

echo \
  "deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/docker.gpg] https://download.docker.com/linux/ubuntu \
  $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null

sudo apt-get --yes update
sudo apt-get --yes install docker-ce docker-ce-cli containerd.io docker-compose-plugin
sudo service docker start
sudo docker compose up -d
